# Через внутреннюю команду Scrapy
scrapy crawl LitnetBooksParser -o books.json -a mode=simple

scrapy crawl LitnetBooksParser -o books.jl -a mode=offset_file #json-lines, будут лучше стыковаться данные при повторной записи в файл
scrapy crawl LitnetBooksParser -o books.json -a mode=offset_file

scrapy crawl LitnetBooksParser -o books.jl -s JOBDIR=crawls/litnet_session -a mode=jobdir #json-lines, будут лучше стыковаться данные при повторной записи в файл
scrapy crawl LitnetBooksParser -o books.json -s JOBDIR=crawls/litnet_session -a mode=jobdir
